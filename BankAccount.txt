import random
name = input ("Name your account. ")
number = random.randint(100000000000000000000000000,900000000000000000000000000)
print ("Your account number is: UA" + str(number))
 
class BankAccount:
   def __init__(account, name):
      account.name = name
      account.balance = 10000
      account.withdraw = 0
      account.deposit = 0
      account.full = True
   def to_spend(account):
     print("Time to spend all my money!!!")
     account.withdraw = random.randint(1,1000) + account.withdraw
     account.balance = account.balance - account.withdraw
 
   def to_work(account):
    print("Time to work...")
    account.balance = account.balance + random.randint(1,2000)
 
   def to_deposit(account):
    print("For the future.")
    account.deposit = random.randint(1,1000) + account.deposit
    account.balance = account.balance - account.deposit
 
   def is_full(account):
    if account.balance < 1:
      print("NOOO WHERE IS MY MONEY")
      if account.deposit > 1:
        print("Wait, I have my deposits!")
        account.balance = account.balance + account.deposit
        account.deposit = 0
      else:
        print("I'm hungry...")
        account.full = False
 
   def end_of_day(account):
    print(f"balance = {account.balance}")
    print(f"withdraw = {round(account.withdraw, 2)}")
    print(f"deposit = {round(account.deposit)}")
 
   def live(account, day):
     day = "Day " + str(day) + " of " + account.name + "'s money management!"
     print(f"{day:=^50}")
     live_cube = random.randint(1, 3)
     if live_cube == 1:
         account.to_spend()
     elif live_cube == 2:
         account.to_work()
     elif live_cube == 3:
         account.to_deposit()
 
     account.end_of_day()
     account.is_full()
 
account = BankAccount(name)
for day in range(20):
    if account.full == False:
        print ("Irresponsible!")
        break
    if account.full == True:
        print ("Next day...")
    account.live(day)
